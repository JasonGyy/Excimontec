34 class Exciton_Hop : public Event{
35  public:
36  static const std::string name;
37  // Singlet FRET hop
38  void calculateExecutionTime(const double prefactor, const double distance, const double E_delta, Simulation* sim_ptr) {
39  double rate = prefactor*Utils::intpow(1.0 / distance, 6);
40  if (E_delta > 0) {
41  rate *= exp(-E_delta / (Utils::K_b*sim_ptr->getTemp()));
42  }
43  Event::calculateExecutionTime(rate, sim_ptr);
44  }
45  // Triplet Dexter hop
46  void calculateExecutionTime(const double prefactor, const double localization, const double distance, const double E_delta, Simulation* sim_ptr) {
47  double rate = prefactor*exp(-2.0*localization*distance);
48  if (E_delta>0) {
49  rate *= exp(-E_delta / (Utils::K_b*sim_ptr->getTemp()));
50  }
51  Event::calculateExecutionTime(rate, sim_ptr);
52  }
53  std::string getName() const{return name;}
54  private:
55 };

64 class Exciton_Dissociation : public Event{
65  public:
66  static const std::string name;
67  void calculateExecutionTime(const double prefactor,const double localization,const double distance,const double E_delta,Simulation* sim_ptr){
68  // Calculates dissociation using the Miller-Abrahams model
69  double rate = prefactor*exp(-2.0*localization*distance);
70  if(E_delta>0){
71  rate *= exp(-E_delta/(Utils::K_b*sim_ptr->getTemp()));
72  }
73  Event::calculateExecutionTime(rate,sim_ptr);
74  }
75  void calculateExecutionTime(const double prefactor,const double localization,const double distance,const double E_delta,const double reorganization,Simulation* sim_ptr){
76  // Calculates dissociation using the Marcus model
77  double rate = (prefactor/sqrt(4.0*Utils::Pi*reorganization*Utils::K_b*sim_ptr->getTemp()))*exp(-2.0*localization*distance)*exp(-Utils::intpow(reorganization+E_delta,2)/(4.0*reorganization*Utils::K_b*sim_ptr->getTemp()));
78  Event::calculateExecutionTime(rate,sim_ptr);
79  }
80  std::string getName() const{return name;}
81  private:
82 };
